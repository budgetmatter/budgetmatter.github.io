    with open('countries.geo.json') as f:
          features = json.load(f)["features"]
          for foo in features:
              try:
                  name = foo['properties']["name"]
                  happydata = happiness.loc[happiness["country"] == name, :]
                  happydict = happydata.to_dict(orient = 'records')[0]
                  foo["properties"] = happydict
                  worldgeojson["features"].append(foo)
              except:
                  print(f'An error occured for name #: {name}')
